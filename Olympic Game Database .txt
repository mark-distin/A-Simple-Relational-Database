Create table OG_Sport
(
	sportNo		integer		not null,
	sportName	text		not null,

	CONSTRAINT OG_SportPK PRIMARY KEY (sportNo),

	CONSTRAINT OG_Sport_No    CHECK ((sportNo >=1 ) and ( sportNo <=15 )),
	CONSTRAINT OG_Sport_Name  CHECK ( sportName IN (
		'alpine skiing', 'biathlon', 'bobsleigh', 'cross country skiing',
		'curling', 'figure skating', 'freestyle skiing', 'ice hockey',  
		'luge', 'nordic combined', 'short track', 'skeleton',  
		'ski jumping', 'snowboard', 'speed skating'))
);

Create table OG_OlympicGame
(
	gameNo		integer		not null,
	gameName	text		not null,
	year		varchar(4)	not null,
	duration	text		not null,
	country		text		not null,
	city		text		not null,
	eventAmount     integer 	not null,
	countryAmount	integer		not null,
	athleteAmount	integer 	not null,

	CONSTRAINT OG_OlympicGamePK PRIMARY KEY (gameNo),

	CONSTRAINT OG_OlympicGame_Name     CHECK (gameName IN (
		'sochi 2014','pyeongchang 2018')),
	CONSTRAINT OG_OlympicGame_Year     CHECK (year IN (
		'2014','2018')),
	CONSTRAINT OG_OlympicGame_Duration CHECK (duration IN (
		'07 feb - 23 feb','09 feb - 25 feb')),
	CONSTRAINT OG_OlympicGame_Country CHECK (country IN (
		'russian federation','republic of korea')),
	CONSTRAINT OG_OlympicGame_City CHECK (city IN (
		'sochi','pyeongchang'))
);


Create table OG_Competition
(
	compNo		integer		not null,
	compName	text		not null,
	eventNo		integer,
	gameNo		integer,


	CONSTRAINT OG_CompetitionPK PRIMARY KEY (compNo),
	CONSTRAINT OG_CompetitionFK_Event       FOREIGN KEY (eventNo) REFERENCES OG_Event
ON DELETE CASCADE
ON UPDATE CASCADE,
	CONSTRAINT OG_CompetitionFK_Olympicgame FOREIGN KEY (gameNo) REFERENCES OG_OlympicGame
ON DELETE CASCADE
ON UPDATE CASCADE,
	
	CONSTRAINT OG_Competition_Compno CHECK ((compNo >= 1) and (compNo <= 160))
);

Create table OG_Athlete
(
	athNo		integer		not null,
	athName		text		not null,
	birth		varchar(4)  	not null,
	countryCode	char(3),
	gender		char(1),

	CONSTRAINT OG_AthletePK PRIMARY KEY (athNo),
	CONSTRAINT OG_AthleteFK FOREIGN KEY (countryCode) REFERENCES OG_Country
ON DELETE CASCADE
ON UPDATE CASCADE,
	CONSTRAINT OG_Athlete_No     CHECK ((athNo >= 1 ) and (athNo < 260)),
	CONSTRAINT OG_Athlete_Birth  CHECK ((birth > 1950) and (birth <2010)),
	CONSTRAINT OG_Athlete_Gender CHECK (gender IN (
		'M',  -- Male
		'F')) -- Female
);



Create table OG_AthleteSport
(
	athNo		integer,	
	sportNo		integer,

	CONSTRAINT OG_AthleteSportPK PRIMARY KEY (athNo, sportNo),
	CONSTRAINT OG_AthleteSportFK_Athlete FOREIGN KEY (athNo) REFERENCES OG_Athlete
ON DELETE CASCADE
ON UPDATE CASCADE,
 	CONSTRAINT OG_AthleteSportFK_Sport   FOREIGN KEY (sportNo) REFERENCES OG_Sport
ON DELETE CASCADE
ON UPDATE CASCADE
);


Create table OG_AthleteOlympic
(
	gameNo		integer,
	athNo		integer,
	
	CONSTRAINT OG_AthleteOlympicPK PRIMARY KEY (gameNo, athNo),
	CONSTRAINT OG_AthleteOlympicFK_OlympicGame FOREIGN KEY (gameNo) REFERENCES OG_OlympicGame
ON DELETE CASCADE
ON UPDATE CASCADE,
	CONSTRAINT OG_AthleteOlympicFK_Athlete     FOREIGN KEY (athNo) REFERENCES OG_Athlete
ON DELETE CASCADE
ON UPDATE CASCADE
);


Create table OG_Result
(
	compNo		integer,
	athNo		integer,
	medal		char(1),

	CONSTRAINT OG_ResultPK PRIMARY KEY (compNo, athNo),
	CONSTRAINT OG_ResultFK_Competition FOREIGN KEY (compNo) REFERENCES OG_Competition
ON DELETE CASCADE
ON UPDATE CASCADE,
	CONSTRAINT OG_ResultFK_Athlete     FOREIGN KEY (athNo) REFERENCES OG_Athlete
ON DELETE CASCADE
ON UPDATE CASCADE,

	CONSTRAINT OG_Result_Medal CHECK ( medal in ( 
		'G', -- Gold
		'S', -- Siliver
		'B'))-- Bronze
);


Create view Alpine_athlete
	(athlete_no, athlete_name, birth, nation, gender) as
	select athNo, athName, birth, countryName, gender
	from  OG_Sport natural join OG_AthleteSport natural join OG_Athlete 
	      natural join OG_Country
	where sportName = 'alpine skiing'
	order by athNo;
